\label{sec:pseudocode}

\subsection{Optimized Hybrid Evolutionary Algorithm (OHEA)}

The Optimized Hybrid Evolutionary Algorithm (OHEA) is a high-performance memetic algorithm designed to maximize solution quality within a fixed time budget. It integrates an aggressive, highly optimized Local Search with a diversity-preserving steady-state evolutionary framework.

Key components of the algorithm include:
\begin{itemize}
    \item \textbf{Fast Local Search:} Utilizes \textit{Candidate Lists} (size 40) to limit neighborhood exploration to promising edges and \textit{Don't Look Bits} (DLB) to avoid redundant checks, significantly speeding up convergence. It performs both intra-route 2-opt moves and inter-route node swaps.
    \item \textbf{Edge Recombination Crossover:} A robust crossover operator that constructs offspring by greedily selecting edges present in either parent. It prioritizes neighbors with fewer remaining connections to preserve edge structure, using spatial distance as a tie-breaker.
    \item \textbf{Diversity Management:} The population (size 20) strictly forbids duplicate objective function values. Offspring only replace the worst individual if they are unique and strictly better.
    \item \textbf{Cataclysmic Restart:} If the global best solution does not improve for a specified number of iterations (300), the population is restarted. The global best is preserved, while other individuals are replaced by new random solutions improved by local search.
\end{itemize}

\begin{algorithm}[H]
    \caption{Optimized Hybrid Evolutionary Algorithm (OHEA)}
    \begin{algorithmic}[1]
        \Require Time Limit $T_{max}$, Population Size $N=20$, Candidates $K=40$
        \Ensure Best Solution $S_{best}$

        \State \textbf{Initialization:}
        \State $Pop \gets \emptyset$
        \While{$|Pop| < N$}
            \State $S \gets \textsc{RandomSolution}()$
            \State $S \gets \textsc{FastLocalSearch}(S, Candidates, DLB)$
            \If{$f(S) \notin \{f(P) \mid P \in Pop\}$}
                \State $Pop \gets Pop \cup \{S\}$
            \EndIf
        \EndWhile
        \State $S_{best} \gets \textsc{GetBest}(Pop)$
        \State $Stagnation \gets 0$

        \State \textbf{Main Loop:}
        \While{$Elapsed(t) < T_{max}$}
            \State \Comment{Selection}
            \State $P_1 \gets \textsc{BinaryTournament}(Pop)$
            \State $P_2 \gets \textsc{Random}(Pop \setminus \{P_1\})$

            \State \Comment{Recombination}
            \State $C \gets \textsc{EdgeRecombinationCrossover}(P_1, P_2)$

            \State \Comment{Mutation (Prob $= 0.05$)}
            \If{$Rand(0,1) < 0.05$}
                \State $C \gets \textsc{DoubleBridgePerturbation}(C)$
            \EndIf

            \State \Comment{Improvement}
            \State $C \gets \textsc{FastLocalSearch}(C, Candidates, DLB)$

            \State \Comment{Steady-State Replacement with Diversity Check}
            \If{$f(C) \notin \{f(P) \mid P \in Pop\}$ \textbf{and} $f(C) < f(\textsc{Worst}(Pop))$}
                \State Replace $\textsc{Worst}(Pop)$ with $C$
                \If{$f(C) < f(S_{best})$}
                    \State $S_{best} \gets C$
                    \State $Stagnation \gets 0$
                \Else
                    \State $Stagnation \gets Stagnation + 1$
                \EndIf
            \Else
                \State $Stagnation \gets Stagnation + 1$
            \EndIf

            \State \Comment{Cataclysm Restart}
            \If{$Stagnation > 300$}
                \State $Pop \gets \{S_{best}\}$
                \While{$|Pop| < N$}
                    \State $S \gets \textsc{RandomSolution}()$
                    \State $S \gets \textsc{FastLocalSearch}(S, Candidates, DLB)$
                    \State $Pop \gets Pop \cup \{S\}$
                \EndWhile
                \State $Stagnation \gets 0$
            \EndIf
        \EndWhile
        \State \Return $S_{best}$
    \end{algorithmic}
\end{algorithm}
